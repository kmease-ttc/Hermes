### Step card component

Each step is a card: badge + title + summary + “Learn more”.

```tsx
type StepProps = {
  label: string;   // A, B, C...
  title: string;
  summary: string;
  href: string;    // link to module page
};

function EngineStep({ label, title, summary, href }: StepProps) {
  return (
    <div className="relative md:pl-16">
      {/* dot on the line (desktop only) */}
      <div className="hidden md:flex absolute left-4 top-6 h-4 w-4 items-center justify-center">
        <span className="h-3 w-3 rounded-full bg-white border border-indigo-400 shadow-[0_0_0_4px_rgba(129,140,248,0.35)]" />
      </div>

      <div className="rounded-2xl bg-white/90 backdrop-blur border border-slate-200 shadow-md hover:shadow-xl hover:-translate-y-0.5 transition flex flex-col gap-3 px-5 py-4 md:px-6 md:py-5">
        <div className="flex items-center gap-3">
          <span className="inline-flex h-7 w-7 items-center justify-center rounded-full bg-indigo-50 text-[11px] font-semibold text-indigo-600">
            {label}
          </span>
          <h3 className="text-sm md:text-base font-semibold text-slate-900">
            {title}
          </h3>
        </div>
        <p className="text-xs md:text-sm text-slate-600">
          {summary}
        </p>
        <div>
          <a
            href={href}
            className="inline-flex items-center gap-1 text-xs font-medium text-indigo-600 hover:text-indigo-700"
          >
            Learn more
            <span>↗</span>
          </a>
        </div>
      </div>
    </div>
  );
}
```
